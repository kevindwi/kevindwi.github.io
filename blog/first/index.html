var (
	queue = make(chan int, 3)
	wg    = &sync.WaitGroup{}
)

func consumer() {
	for {
		task, open := <-queue
		if !open {
			break
		}
		wg.Add(1)
		go func(val int) {
			defer wg.Done()
			// some logic
		}(task)
	}
}

func producer() {
	var arr []int
	for i := 0; i < 1000000; i++ {
		arr = append(arr, i)
	}
	for _, val := range arr {
		select {
		case queue <- val:
		case <-time.After(time.Millesecond):
			fmt.Print("producer timeout")
		}
	}
}

func main() {
	go consumer()
	producer()
	wg.Wait()
}
